CREATE TABLE PHOTO(
    NO NUMBER,
    TITLE VARCHAR2(300),
    CONTENT VARCHAR2(300),
    READ_COUNT NUMBER,
    WRITE_DATE DATE,
    USER_ID VARCHAR2(300)
);



CREATE TABLE PHOTO_REPLY(
    NO NUMBER,
    PHOTO_NO NUMBER,
    WRITER VARCHAR2(300),
    CONTENT VARCHAR2(300),
    WRITE_DATE DATE
);
